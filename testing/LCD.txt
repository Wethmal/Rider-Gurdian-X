#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// Change 0x27 to 0x3F if your LCD uses that address
LiquidCrystal_I2C lcd(0x27, 16, 2);

void setup() {
  Wire.begin();       // For ESP32: Wire.begin(21, 22);
  lcd.init();         // Initialize LCD
  lcd.backlight();    // Turn on backlight

  lcd.setCursor(0, 0);
  lcd.print("LCD Test");
  lcd.setCursor(0, 1);
  lcd.print("Working!");
  delay(2000);
}

void loop() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Count: ");
  
  static int count = 0;
  lcd.print(count);
  count++;

  delay(1000);
}
