#include <SoftwareSerial.h>

// -----------------------
// Ultrasonic Pins
// -----------------------
#define TRIG_PIN D6
#define ECHO_PIN D7

// -----------------------
// Bluetooth Pins
// -----------------------
SoftwareSerial BT(D3, D8);  // HC-05: D3 = RX, D8 = TX

// -----------------------
// Setup
// -----------------------
void setup() {
  Serial.begin(115200);
  BT.begin(9600);

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  Serial.println("Ultrasonic + Bluetooth Ready");
  BT.println("Ultrasonic + Bluetooth Ready");
}

// -----------------------
// Distance Function
// -----------------------
long getDistance() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);

  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  long duration = pulseIn(ECHO_PIN, HIGH);
  long distance = duration * 0.034 / 2;   // Convert to cm

  return distance;
}

// -----------------------
// Main Loop
// -----------------------
void loop() {

  long distance = getDistance();

  // Print to Serial
  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");

  // Send to Bluetooth
  BT.print("Distance: ");
  BT.print(distance);
  BT.println(" cm");

  // Optional warning
  if (distance < 20) {
    Serial.println("⚠ Object Close!");
    BT.println("⚠ Object Close!");
  }

  delay(500);
}
