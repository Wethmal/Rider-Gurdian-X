package com.example.ridergurdianx;

import android.app.NotificationChannel;
import android.app.NotificationManager;
import android.content.Context;
import android.content.SharedPreferences;
import android.os.Build;
import android.os.Bundle;
import android.text.InputType;
import android.util.Log;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import android.widget.Toast;

import androidx.annotation.NonNull;
import androidx.appcompat.app.AlertDialog;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.app.NotificationCompat;
import androidx.core.app.NotificationManagerCompat;

import com.google.firebase.database.DataSnapshot;
import com.google.firebase.database.DatabaseError;
import com.google.firebase.database.DatabaseReference;
import com.google.firebase.database.FirebaseDatabase;
import com.google.firebase.database.ValueEventListener;

public class UltrasonicActivity extends AppCompatActivity {

    private static final String TAG = "UltrasonicActivity";

    private TextView tvDistance;
    private TextView tvStatus;
    private TextView tvAdvice;
    private TextView tvSafeDistance;
    private Button btnChangeSafeDistance;

    private DatabaseReference ultrasonicRef;

    // prefs
    private static final String PREFS_NAME = "rider_guardian_prefs";
    private static final String KEY_SAFE_DISTANCE_CM = "safe_distance_cm";
    private static final int DEFAULT_SAFE_DISTANCE_CM = 80;

    // notification
    private static final String CHANNEL_ID = "ultra_alert_channel";
    private static final int NOTIFICATION_ID = 2001;

    private int safeDistanceCm = DEFAULT_SAFE_DISTANCE_CM;
    private boolean unsafeNotified = false;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_ultrasonic);

        // ---- bind views (MATCH XML IDs ABOVE) ----
        tvDistance       = findViewById(R.id.tvUltrasonicDistance);
        tvStatus         = findViewById(R.id.tvUltrasonicStatus);
        tvAdvice         = findViewById(R.id.tvUltrasonicAdvice);
        tvSafeDistance   = findViewById(R.id.tvSafeDistance);
        btnChangeSafeDistance = findViewById(R.id.btnChangeSafeDistance);

        // ---- load saved safe distance ----
        SharedPreferences prefs =
                getSharedPreferences(PREFS_NAME, Context.MODE_PRIVATE);
        safeDistanceCm = prefs.getInt(KEY_SAFE_DISTANCE_CM, DEFAULT_SAFE_DISTANCE_CM);
        updateSafeDistanceLabel();

        btnChangeSafeDistance.setOnClickListener(v -> showSafeDistanceDialog());

        createNotificationChannel();

        // ---- connect to your exact DB instance + node "ultrasonic" ----
        FirebaseDatabase db = FirebaseDatabase.getInstance(
                "https://iottest-6f78a-default-rtdb.asia-southeast1.firebasedatabase.app/"
        );

        ultrasonicRef = db.getReference("ultrasonic");

        // sanity log: see whole snapshot
        ultrasonicRef.addValueEventListener(new ValueEventListener() {
            @Override
            public void onDataChange(@NonNull DataSnapshot snapshot) {

                Log.d(TAG, "onDataChange: snapshot = " + snapshot);

                if (!snapshot.exists()) {
                    tvDistance.setText("Distance: -- cm");
                    tvStatus.setText("Status: --");
                    tvAdvice.setText("No ultrasonic data available.");
                    return;
                }

                Object distObj   = snapshot.child("distance").getValue();
                Object statusObj = snapshot.child("status").getValue();

                Log.d(TAG, "distance raw = " + distObj + ", status raw = " + statusObj);

                if (distObj == null) {
                    tvDistance.setText("Distance: -- cm");
                    tvStatus.setText("Status: --");
                    tvAdvice.setText("Waiting for sensor data...");
                    return;
                }

                double distanceCm;
                try {
                    distanceCm = Double.parseDouble(distObj.toString());
                } catch (NumberFormatException e) {
                    Log.e(TAG, "Invalid distance value: " + distObj, e);
                    distanceCm = -1;
                }

                if (distanceCm < 0) {
                    tvDistance.setText("Distance: -- cm");
                    tvStatus.setText("Status: --");
                    tvAdvice.setText("Invalid distance value.");
                    return;
                }

                tvDistance.setText(String.format("Distance: %.2f cm", distanceCm));

                String statusText;
                if (statusObj != null) {
                    statusText = statusObj.toString();
                } else {
                    statusText = (distanceCm >= safeDistanceCm) ? "SAFE" : "UNSAFE";
                }

                tvStatus.setText("Status: " + statusText);

                if (distanceCm >= safeDistanceCm) {
                    tvAdvice.setText("Safe distance from obstacle.");
                    unsafeNotified = false;   // reset
                } else {
                    tvAdvice.setText("Too close! Slow down or brake immediately.");

                    if (!unsafeNotified) {
                        sendUnsafeNotification(distanceCm);
                        unsafeNotified = true;
                    }
                }
            }

            @Override
            public void onCancelled(@NonNull DatabaseError error) {
                Log.e(TAG, "onCancelled: " + error.getMessage());
                Toast.makeText(UltrasonicActivity.this,
                        "Failed to load ultrasonic data: " + error.getMessage(),
                        Toast.LENGTH_SHORT).show();
            }
        });
    }

    // -------- Safe distance UI --------

    private void updateSafeDistanceLabel() {
        tvSafeDistance.setText("Safe Distance: " + safeDistanceCm + " cm");
    }

    private void showSafeDistanceDialog() {
        final EditText input = new EditText(this);
        input.setInputType(InputType.TYPE_CLASS_NUMBER);
        input.setHint("Enter safe distance in cm");
        input.setText(String.valueOf(safeDistanceCm));

        new AlertDialog.Builder(this)
                .setTitle("Set Safe Distance")
                .setMessage("Distance ABOVE this value will be SAFE. Below it will trigger an alert.")
                .setView(input)
                .setPositiveButton("Save", (dialog, which) -> {
                    String txt = input.getText().toString().trim();
                    if (!txt.isEmpty()) {
                        try {
                            int value = Integer.parseInt(txt);
                            if (value <= 0) throw new NumberFormatException();
                            safeDistanceCm = value;
                            SharedPreferences prefs =
                                    getSharedPreferences(PREFS_NAME, Context.MODE_PRIVATE);
                            prefs.edit().putInt(KEY_SAFE_DISTANCE_CM, safeDistanceCm).apply();
                            updateSafeDistanceLabel();
                            unsafeNotified = false;
                            Toast.makeText(this,
                                    "Safe distance updated to " + safeDistanceCm + " cm",
                                    Toast.LENGTH_SHORT).show();
                        } catch (NumberFormatException e) {
                            Toast.makeText(this,
                                    "Invalid number.",
                                    Toast.LENGTH_SHORT).show();
                        }
                    }
                })
                .setNegativeButton("Cancel", null)
                .show();
    }

    // -------- Notification --------

    private void createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            NotificationChannel channel = new NotificationChannel(
                    CHANNEL_ID,
                    "Ultrasonic Alerts",
                    NotificationManager.IMPORTANCE_HIGH
            );
            NotificationManager manager = getSystemService(NotificationManager.class);
            if (manager != null) {
                manager.createNotificationChannel(channel);
            }
        }
    }

    private void sendUnsafeNotification(double distanceCm) {
        String message = "Obstacle detected at "
                + String.format("%.1f", distanceCm)
                + " cm (Safe â‰¥ " + safeDistanceCm + " cm).";

        NotificationCompat.Builder builder =
                new NotificationCompat.Builder(this, CHANNEL_ID)
                        .setSmallIcon(R.mipmap.ic_launcher)
                        .setContentTitle("Unsafe Distance Alert")
                        .setContentText(message)
                        .setPriority(NotificationCompat.PRIORITY_HIGH)
                        .setAutoCancel(true);

        NotificationManagerCompat manager = NotificationManagerCompat.from(this);
        manager.notify(NOTIFICATION_ID, builder.build());
    }
}
